``` r
simfun = function(samp_size = 30, mu, sigma = 5){
  
  prob3_data = 
    tibble(
      x = rnorm(n = samp_size, mean = mu, sd = sigma)
    ) 
  
   ttest = t.test(prob3_data, conf.level = .95) %>% broom::tidy()

   return(ttest)
}
```

``` r
prob3_result0 =
```
